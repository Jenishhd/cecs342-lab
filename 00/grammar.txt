<program>   ::= <stmt>

<stmt>      ::= <let>
<stmt>      ::= <print>
<stmt>      ::= <stmt> SEMICOLON <stmt>

<let>       ::= LET IDENTIFIER EQ <expr>

<expr>      ::= NUM
<expr>      ::= iden

<print>     ::= PRINT IDENTIFIER

EQ: =
IDENTIFIER: [a-zA-Z][a-zA-Z0-9_]+
LET: let
NUM: [0-9]+
PRINT: print
SEMICOLON: ;
